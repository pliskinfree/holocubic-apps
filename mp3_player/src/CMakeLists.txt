cmake_minimum_required(VERSION 3.16)

if(CMAKE_MAKE_PROGRAM AND EXISTS "${CMAKE_MAKE_PROGRAM}")
    set(_audio_ninja_cmd "${CMAKE_MAKE_PROGRAM}")
elseif(DEFINED ENV{NINJA} AND EXISTS "$ENV{NINJA}")
    set(_audio_ninja_cmd "$ENV{NINJA}")
elseif(DEFINED ENV{IDF_TOOLS_PATH})
    set(_audio_ninja_candidate "$ENV{IDF_TOOLS_PATH}/tools/ninja/ninja.exe")
    if(EXISTS "${_audio_ninja_candidate}")
        set(_audio_ninja_cmd "${_audio_ninja_candidate}")
    endif()
endif()

if(_audio_ninja_cmd)
    file(TO_CMAKE_PATH "${_audio_ninja_cmd}" _audio_ninja_cmd)
    set(CMAKE_MAKE_PROGRAM "${_audio_ninja_cmd}" CACHE FILEPATH "Path to ninja executable" FORCE)
    set(ENV{CMAKE_MAKE_PROGRAM} "${_audio_ninja_cmd}")
    set(ENV{NINJA} "${_audio_ninja_cmd}")
    get_filename_component(_audio_ninja_dir "${_audio_ninja_cmd}" DIRECTORY)
    if(_audio_ninja_dir)
        set(ENV{PATH} "${_audio_ninja_dir};$ENV{PATH}")
    endif()
    set(_audio_extra_cmake_arg "-DCMAKE_MAKE_PROGRAM=${_audio_ninja_cmd}")
else()
    set(_audio_extra_cmake_arg "")
endif()

include($ENV{IDF_PATH}/tools/cmake/project.cmake)
if(_audio_extra_cmake_arg)
    idf_build_set_property(EXTRA_CMAKE_ARGS "${_audio_extra_cmake_arg}" APPEND)
endif()

project(audio)
